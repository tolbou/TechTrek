'''
# 条件式のコード例
# if文
age = 25

if age >= 18
  puts "成人です"
elsif age >= 25
	puts "アラサーです" 
else
  puts "未成年です"
end

# unless文
is_student = true

unless is_student
  puts "学生ではありません"
else
  puts "学生です"
end

# case文
fruit = "apple"

case fruit
when "apple"
  puts "りんごです"
when "banana"
  puts "バナナです"
else
  puts "その他の果物です"
end
'''
